// Függvény definíciója a két kártya húzására a 52 lapos francia kártyából
var húzás = function () {
  // Véletlenszerűen választunk két kártyát a 52 lapos pakliból
  var huzas1 = randomInteger(52);
  var huzas2 = randomInteger(52);
  // Feltétel: a két kártya legyen különböző 
  condition(huzas1 !== huzas2);
  // A két húzott kártya visszaadása
  return [huzas1, huzas2];
}

// Függvény definíciója a kedvező kártya húzására: az egyik király, a másik nem király
var kedvező_húzás = function () {
  // Véletlenszerűen választunk két kártyát a 52 lapos pakliból
  var huzas1 = randomInteger(52);
  var huzas2 = randomInteger(52);
  // Feltétel: a két kártya legyen különböző, és az egyik legyen király, a másik nem király
  condition((huzas1 !== huzas2) && ((huzas1 < 4 && huzas2 >= 4) || (huzas2 < 4 && huzas1 >= 4)));
  // A két húzott kártya 
  return [huzas1, huzas2];
}

// Kedvező kártya húzásának kiszámolása enumerációval
var kedvező = Enumerate(kedvező_húzás);
// Kedvező kártyák gyakoriságának megjelenítése hisztogramon
viz.hist(kedvező);

// Valószínűség kiszámítása: kedvező esetek aránya az összes esethez képest
print("p = " + kedvező.supp.length / összes.supp.length);
