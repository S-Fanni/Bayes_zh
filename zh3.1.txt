var húzás = function () {
  var huzas1 = randomInteger(52)
  var huzas2 = randomInteger(52)
  condition(huzas1 !== huzas2)
  return [huzas1, huzas2];
}

var kedvező_húzás = function () {
  var huzas1 = randomInteger(52)
  var huzas2 = randomInteger(52)
  condition(huzas1 !== huzas2 && huzas1 < 4 && huzas2 >= 4 || huzas2 < 4 && huzas1 >= 4)
  return  [huzas1,huzas2];
}
